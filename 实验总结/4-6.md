问题描述如下：

/*
输入数组a( 长度为n, n<10)，对元素按由小到大顺序排列，然后再输入一个数b插入数组a中，插入后，数组a中的元素仍然由小到大顺序排列。
输入(第一行数组大小n,第二行为输入的数组，第三行为插入的整数):
5
5 2 3 4 8
7
输出：
2 3 4 5 7 8
*/

样例代码如下：
```
#include <stdio.h>
#include <string.h>

#define MAX 10

//输入数组
void printArr(int *arr, int len)
{
    int i ;
    for (i = 0; i < len; ++i)
    {
        printf("%d ",arr[i]);   
    }
    
}

//对数组进行排序
void sort(int *arr, int len)
{
    int i, j;
    int temp ;
    for(i = 0 ; i < len ; i ++)
    {
        for (j = i + 1; j < len; j ++)
        {
            if(arr[i] > arr[j])
            {
                temp = arr[j];
                arr[j] = arr[i];
                arr[i] = temp;
                
            }
        }
    }
}

//插入元素到数组
void insert(int *arr, int index, int insertNum, int len)
{
    int i;
    for(i = len; i > index; i--)
    {
        arr[i] = arr[i-1];
    }
    arr[i] = insertNum;
}

//获取要插入的位置
int get_index(int *arr, int insertNum, int len)
{
    int i;
    int index = len;
    for(i = 0 ; i < len ; i++)
    {
        if(insertNum <= arr[i])
        {
            index = i;
            break;
        }
    }
    return index;
}

int main()
{
    int arr[10];
    int n;
    int i ;
    int insertNum ;
    int insert_index;
    scanf("%d", &n);
    for (i = 0 ; i < n ; i++)
    {
    	scanf("%d", &arr[i]);
    }
    scanf("%d", &insertNum);
    
    sort(arr, n);
    
    insert_index = get_index(arr, insertNum, n);
    insert(arr, insert_index, insertNum, n);
    
    printArr(arr, n+1);
    
    return 0;
}
```
